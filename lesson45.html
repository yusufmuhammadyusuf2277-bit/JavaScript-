JavaScript Animations

Animate elements for better UX using JavaScript or CSS: - setInterval / requestAnimationFrame - Change position, size, opacity - CSS transitions / animations 

let box = document.getElementById('box');
let pos = 0;
let id = setInterval(frame, 10);
function frame(){
  if(pos==350){ clearInterval(id); }
  else { pos++; box.style.left = pos+'px'; }
}
